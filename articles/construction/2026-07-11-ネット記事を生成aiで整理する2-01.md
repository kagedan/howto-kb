def load(f):
    try:
        return json.load(open(f, encoding="utf-8"))
    except Exception:
        return None

def text_of(d):
    title=str(d.get("file",""))
    summary=str(d.get("summary",""))
    kws=d.get("keywords") or []
    if isinstance(kws,str): kws=[kws]
    kwtext=" ".join(str(k) for k in kws)
    return title+" "+summary+" "+(kwtext+" ")*3, kwtext, title

def score(text):
    return {name: sum(text.count(kw) for kw in kws) for name,kws in CATS}

def classify():
    files=sorted(f for f in glob.glob(os.path.join(SRC,"*.json"))
                 if os.path.basename(f) != "分類結果.json")
    rows=[]; counter=collections.Counter()
    for f in files:
        data=load(f)
        if data is None:
            rows.append([os.path.basename(f),"","99_その他・未分類",""]); counter["99_その他・未分類"]+=1; continue
        items=data if isinstance(data,list) else [data]
        for d in items:
            text,kwtext,title=text_of(d); sc=score(text); best=max(sc.values())
            if best==0: cat="99_その他・未分類"
            else:
                for name,_ in CATS:
                    if sc[name]==best: cat=name; break
            counter[cat]+=1
            rows.append([os.path.basename(f), title, cat, kwtext])
    return files,rows,counter

def save_markdown(rows, counter, outpath):
    cat_rows = collections.defaultdict(list)
    for row in rows:
        cat_rows[row[2]].append(row)

    lines = ["# 分類結果レポート", ""]
    lines += ["## サマリー", ""]
    lines += ["| カテゴリ | 件数 |", "|---|---|"]
    for k, v in sorted(counter.items()):
        lines.append(f"| {k} | {v} |")
    lines += ["", f"**総数: {len(rows)} 件**", ""]

    lines += ["## カテゴリ別一覧", ""]
    for cat, _ in CATS + [("99_その他・未分類", [])]:
        group = cat_rows.get(cat, [])
        if not group:
            continue
        lines += [f"### {cat} （{len(group)} 件）", ""]
        lines += ["| バッチファイル | 記事ファイル名 | キーワード |", "|---|---|---|"]
        for row in group:
            fname, title, _, kwtext = row
            kw_short = kwtext[:60] + "…" if len(kwtext) > 60 else kwtext
            lines.append(f"| {fname} | {title} | {kw_short} |")
        lines.append("")

    with open(outpath, "w", encoding="utf-8") as f:
        f.write("\n".join(lines))
    print(f"Markdown保存: {outpath}")

if __name__=="__main__":

    save_markdown(rows, counter, md_path)
