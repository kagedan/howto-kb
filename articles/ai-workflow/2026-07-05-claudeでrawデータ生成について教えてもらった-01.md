# rawデータ生成方法をClaudeに聞いてみた

## 背景

raw現像などするとき，各社が提供するデバイスでrawデータ異なるためどのようなものか調べてみたくなった。
作ったアプリはすべてClaudeに作成してもらった．

## 作成するアプリ

16bitで1画素とし，ビッグエンディアンでrawデータを生成する．
コードはC言語で実装し，Pythonで表示できるようにしてみる．

### increment rawデータ生成

1から順で画素の値をインクリメントするように生成する．

```cpp
// rawinccre.cpp
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <iostream>
using namespace std;

int main( int argc, char *argv[] )
{
  const char *fname  = argv[1];
  int         width  = atoi(argv[2]);
  int         height = atoi(argv[3]);
  FILE       *fp;
  int         x;
  int         y;

  /* 使い方: ./make_be [出力ファイル] [幅] [高さ] */
  if( argc>=2 ) {
    fname = argv[1];
  }
  if( argc>=4 ) {
    width  = atoi(argv[2]);
    height = atoi(argv[3]);
  }
  if( width<=0||height<=0 ) {
    fprintf(stderr, "invalid size: %dx%d\n", width, height);
    return 1;
  }

  fp = fopen(fname, "wb");
  if( fp==NULL ) {
    fprintf(stderr, "cannot open: %s\n", fname);
    return 1;
  }

  for( y=0; y<height; y++ ) {
    for( x=0; x<width; x++ ) {
      uint16_t pix = (uint16_t)(y*width+x+1); /* テストパターン: 1,2,3,... */
      uint8_t  buf[2];

      buf[0] = (uint8_t)((pix>>8)&0xFF); /* 上位バイトを先(低位アドレス)へ */
      buf[1] = (uint8_t)(pix&0xFF);      /* 下位バイトを後(高位アドレス)へ */

      if( fwrite(buf,1,2,fp)!=2 ) {
        fprintf(stderr, "write error\n");
        fclose(fp);
        return 1;
      }
    }
  }

  fclose(fp);
  printf("%s : %dx%d, 16bit/pixel, big-endian, %d bytes\n",
         fname, width, height, width*height*2);
  return 0;
}
```

```sh
g++ rawinccre.cpp -o rawinccre
rawinccre sample.bin 100 100
```

これでraw画像が生成される．

### change_csv_to_raw

次に，csvからrawに変換する．

```cpp
// change_csv_to_raw.cpp
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <ctype.h>
#include <iostream>
using namespace std;

#define TOKEN_MAX 64

static int write_pixel_be( FILE *fp, uint16_t pix )
{
  uint8_t buf[2];

  buf[0] = (uint8_t)((pix>>8)&0xFF); /* 上位バイトを先(低位アドレス)へ */
  buf[1] = (uint8_t)(pix&0xFF);      /* 下位バイトを後(高位アドレス)へ */

  if( fwrite(buf,1,2,fp)!=2 ) {
    return -1;
  }
  return 0;
}

static int parse_token( const char *token, long line_no, uint16_t *pix )
{
  const char *p    = token;
  char       *endp;
  long        val;
  int         base = 10;

  while( isspace((unsigned char)*p) ) {
    p++;
  }
  if( p[0]=='0'&&(p[1]=='x'||p[1]=='X') ) {
    base = 16; /* 0x接頭辞のみ16進。base=0は"0010"が8進扱いされる罠があるので使わない */
  }

  val = strtol(p, &endp, base);
  while( *endp!='\0'&&isspace((unsigned char)*endp) ) {
    endp++;
  }
  if( endp==p||*endp!='\0' ) {
    fprintf(stderr, "line %ld: invalid token \"%s\"\n", line_no, token);
    return -1;
  }
  if( val<0||val>0xFFFF ) {
    fprintf(stderr, "line %ld: out of 16bit range(0-65535): %ld\n", line_no, val);
    return -1;
  }

  *pix = (uint16_t)val;
  return 0;
}

int main( int argc, char *argv[] )
{
  const char *in_name  = "sample.csv";
  const char *out_name = "raw.bin";
  FILE       *ifp;
  FILE       *ofp;
  char        token[TOKEN_MAX];
  int         tlen      = 0;
  int         c;
  long        width     = 0; /* 1ライン目の画素数で確定 */
  long        cur_width = 0; /* 現在ラインの画素数 */
  long        height    = 0;
  long        line_no   = 1;
  long        total     = 0;

  /* 使い方: ./csv2be [入力csv] [出力bin] */
  if( argc>=2 ) {
    in_name = argv[1];
  }
  if( argc>=3 ) {
    out_name = argv[2];
  }

  ifp = fopen(in_name, "r");
  if( ifp==NULL ) {
    fprintf(stderr, "cannot open input: %s\n", in_name);
    return 1;
  }

  ofp = fopen(out_name, "wb");
  if( ofp==NULL ) {
    fprintf(stderr, "cannot open output: %s\n", out_name);
    fclose(ifp);
    return 1;
  }

  while( 1 ) {
    c = fgetc(ifp);

    if( c==','||c=='\n'||c==EOF ) {
      token[tlen] = '\0';

      if( tlen>0 ) {
        uint16_t pix;

        if( parse_token(token, line_no, &pix)!=0 ) {
          goto error;
        }
        if( write_pixel_be(ofp, pix)!=0 ) {
          fprintf(stderr, "write error\n");
          goto error;
        }
        cur_width++;
        total++;
      } else if( c==',' ) {
        fprintf(stderr, "line %ld: empty field\n", line_no);
        goto error;
      }
      tlen = 0;

      if( c=='\n'||c==EOF ) {
        if( cur_width>0 ) {
          if( width==0 ) {
            width = cur_width;
          } else if( cur_width!=width ) {
            fprintf(stderr, "line %ld: width mismatch(%ld pixels, expected %ld)\n",
                    line_no, cur_width, width);
            goto error;
          }
          height++;
          cur_width = 0;
        }
        if( c==EOF ) {
          break;
        }
        line_no++;
      }
    } else if( c=='\r' ) {
      /* CRLF対策: CRは読み飛ばす(Windowsで編集したCSVも可) */
    } else {
      if( tlen>=TOKEN_MAX-1 ) {
        fprintf(stderr, "line %ld: token too long\n", line_no);
        goto error;
      }
      token[tlen] = (char)c;
      tlen++;
    }
  }

  if( total==0 ) {
    fprintf(stderr, "no pixel data: %s\n", in_name);
    goto error;
  }

  fclose(ifp);
  fclose(ofp);
  printf("%s -> %s : %ldx%ld, 16bit/pixel, big-endian, %ld bytes\n",
         in_name, out_name, width, height, total*2);
  return 0;

error:
  fclose(ifp);
  fclose(ofp);
  remove(out_name); /* 失敗時は不完全な出力binを残さない */
  return 1;
}
```

以下で，コンパイルし実行できる．

```sh
g++ change_csv_to_raw.cpp -o change_csv_to_raw
change_csv_to_raw sample.csv
```

これでcsvからrawを生成できる．
csvは画像のサイズで改行されている前提ではあるがrawは改行を気にしない．

### raw viewer

次にrawをGUIで見えるようにする．
tkinterがインストールしてあること．

```py
#!/usr/bin/env python3

import sys
import math
import array
import tkinter as tk

SCALES = [(1,16),(1,8),(1,4),(1,2),(1,1),(2,1),(4,1),(8,1),(16,1),(32,1),(64,1),(128,1)]
VAL_MIN_SCALE   = 32   # 画素値をセル内表示する最小倍率
COORD_MIN_SCALE = 64   # x,y座標も表示する最小倍率
XOR80   = bytes((b^0x80) for b in range(256))  # signed用: 上位バイト変換テーブル
DIFF_EQ = bytes((16, 16, 16))                  # diff表示: 一致画素の色(暗灰)


def load_bin( path, w, h, sign ):
  n = w*h
  with open(path, "rb") as f:
    data = f.read()
  if( len(data)!=n*2 ):
    sys.exit("size mismatch: %s is %d bytes, expected %d bytes(%dx%dx2)"
             % (path, len(data), n*2, w, h))
  raw = array.array("H")
  raw.frombytes(data)
  if( sys.byteorder=="little" ):
    raw.byteswap()  # ファイルはビッグエンディアン固定
  if( sign==1 ):
    vals = array.array("i", (u-0x10000 if u>=0x8000 else u for u in raw))
  else:
    vals = array.array("i", raw)
  return data, raw, vals


def make_jet_lut():
  lut  = []
  luma = []
  for i in range(256):
    t = i/255.0
    r = int(255*min(max(1.5-abs(4.0*t-3.0), 0.0), 1.0))
    g = int(255*min(max(1.5-abs(4.0*t-2.0), 0.0), 1.0))
    b = int(255*min(max(1.5-abs(4.0*t-1.0), 0.0), 1.0))
    lut.append(bytes((r, g, b)))
    luma.append(int(0.299*r+0.587*g+0.114*b))
  return lut, luma


def base_name( path, w, h, sign ):
  return "%s  %dx%d 16bit BE %s" % (path, w, h, "signed" if sign==1 else "unsigned")


class Sync:
  # 複数ウィンドウのズーム/パン/マーカー同期
  def __init__( self ):
    self.wins = []
    self.busy = False

  def register( self, w ):
    self.wins.append(w)

  def push( self, src ):
    if( self.busy ):
      return
    self.busy = True
    for w in self.wins:
      if( w is not src ):
        w.si   = src.si
        w.vx   = src.vx
        w.vy   = src.vy
        w.mark = src.mark
        w.redraw()
        if( w.mark is not None ):
          w.status.config(text=w.fmt_status(w.mark[0], w.mark[1]))
    self.busy = False


class Viewer(tk.Toplevel):
  def __init__( self, master, name, w, h, sign, data, raw, vals
              , maxc=None, geom=None, sync=None ):
    super().__init__(master)
    self.iw   = w
    self.ih   = h
    self.sign = sign
    self.data = data
    self.raw  = raw
    self.vals = vals
    self.sync = sync
    self.vmin = min(vals)
    self.vmax = max(vals)
    self.lut, self.lut_luma = make_jet_lut()
    self.cmode = 0     # 0:grayscale 1:pseudo-color
    self.rmode = 0     # 0:16bit固定レンジ 1:データmin-max自動
    self.mark  = None
    self.gray  = b""
    self.rebuild_gray()
    self.set_title(name)

    self.base = None
    self.build_base()
    self.view = None
    self.si   = SCALES.index((1,1))
    self.vx    = 0.0
    self.vy    = 0.0
    self.drag  = None
    self.moved = False
    self.init  = False

    bar = tk.Frame(self)
    bar.pack(side="top", fill="x")
    self.build_toolbar(bar)

    self.status = tk.Label(self, anchor="w", relief="sunken")
    self.status.pack(side="bottom", fill="x")

    sw = self.winfo_screenwidth()
    sh = self.winfo_screenheight()
    if( maxc is None ):
      maxc = (sw-64, sh-160)
    cw = min(self.iw, maxc[0])
    ch = min(self.ih, maxc[1])
    self.canvas = tk.Canvas(self, width=cw, height=ch, bg="#404040",
                            highlightthickness=0, cursor="crosshair")
    self.canvas.pack(side="top", fill="both", expand=True)

    self.canvas.bind("<Configure>",       self.on_configure)
    self.canvas.bind("<ButtonPress-1>",   self.on_press)
    self.canvas.bind("<B1-Motion>",       self.on_drag)
    self.canvas.bind("<ButtonRelease-1>", self.on_release)
    self.canvas.bind("<Motion>",          self.on_motion)
    self.canvas.bind("<MouseWheel>",      self.on_wheel)                         # Windows
    self.canvas.bind("<Button-4>",        lambda e: self.zoom_at(e.x, e.y, +1))  # X11
    self.canvas.bind("<Button-5>",        lambda e: self.zoom_at(e.x, e.y, -1))
    self.bind("<Key-plus>",   lambda e: self.zoom_step(+1))
    self.bind("<Key-minus>",  lambda e: self.zoom_step(-1))
    self.bind("<Key-f>",      lambda e: self.zoom_fit())
    self.bind("<Key-c>",      lambda e: self.toggle_color())
    self.bind("<Key-Escape>", lambda e: self.clear_mark())
    self.bind("<Key-q>",      lambda e: self.quit_all())

    if( geom is not None ):
      self.geometry(geom)
    self.protocol("WM_DELETE_WINDOW", self.quit_all)
    if( sync is not None ):
      sync.register(self)

  def quit_all( self ):
    self.master.destroy()

  def set_title( self, name ):
    self.title(name+"  val[%d..%d]" % (self.vmin, self.vmax))

  # ---- ツールバー -----------------------------------------------------------
  def add_zoom_buttons( self, bar ):
    tk.Button(bar, text="-",   command=lambda: self.zoom_step(-1)).pack(side="left")
    tk.Button(bar, text="+",   command=lambda: self.zoom_step(+1)).pack(side="left")
    tk.Button(bar, text="1:1", command=self.zoom_1to1).pack(side="left")
    tk.Button(bar, text="Fit", command=self.zoom_fit).pack(side="left")

  def add_xy_entry( self, bar ):
    self.zlabel = tk.Label(bar, text="")
    self.zlabel.pack(side="left", padx=8)
    tk.Label(bar, text=" x:").pack(side="left")
    self.xent = tk.Entry(bar, width=6)
    self.xent.pack(side="left")
    tk.Label(bar, text=" y:").pack(side="left")
    self.yent = tk.Entry(bar, width=6)
    self.yent.pack(side="left")
    tk.Button(bar, text="Go", command=self.goto_xy).pack(side="left", padx=4)
    self.xent.bind("<Return>", lambda e: self.goto_xy())
    self.yent.bind("<Return>", lambda e: self.goto_xy())

  def build_toolbar( self, bar ):
    self.add_zoom_buttons(bar)
    self.cbtn = tk.Button(bar, text="Color", command=self.toggle_color)
    self.cbtn.pack(side="left", padx=4)
    self.rbtn = tk.Button(bar, text="Range:16bit", command=self.toggle_range)
    self.rbtn.pack(side="left", padx=4)
    self.add_xy_entry(bar)
    self.lframe = tk.Frame(bar)
    self.llo = tk.Label(self.lframe, text="")
    self.llo.pack(side="left")
    lc = tk.Canvas(self.lframe, width=130, height=14,
                   highlightthickness=1, highlightbackground="#808080")
    for i in range(128):
      c3 = self.lut[i*2]
      lc.create_line(i+2, 1, i+2, 14, fill="#%02x%02x%02x" % (c3[0], c3[1], c3[2]))
    lc.pack(side="left", padx=2)
    self.lhi = tk.Label(self.lframe, text="")
    self.lhi.pack(side="left")
    self.update_legend()

  # ---- 表示用8bit値(gray)の生成 --------------------------------------------
  def rebuild_gray( self ):
    if( self.rmode==0 ):
      # 16bit固定レンジ: BEファイルの上位バイト列がそのまま v>>8 になる
      hi = self.data[0::2]
      if( self.sign==1 ):
        self.gray = hi.translate(XOR80)  # (v+32768)>>8 と等価
      else:
        self.gray = hi
    else:
      vmin = self.vmin
      vmax = self.vmax
      if( vmax>vmin ):
        rng = vmax-vmin
        self.gray = bytes(((v-vmin)*255)//rng for v in self.vals)
      else:
        self.gray = bytes(128 for _ in self.vals)

  def range_labels( self ):
    if( self.rmode==0 ):
      if( self.sign==1 ):
        return -32768, 32767
      return 0, 65535
    return self.vmin, self.vmax

  def update_legend( self ):
    lo, hi = self.range_labels()
    self.llo.config(text=str(lo))
    self.lhi.config(text=str(hi))

  def build_base( self ):
    if( self.cmode==0 ):
      img = b"P5\n%d %d\n255\n" % (self.iw, self.ih)+self.gray
    else:
      img = b"P6\n%d %d\n255\n" % (self.iw, self.ih) \
            +b"".join(self.lut[g] for g in self.gray)
    self.base = tk.PhotoImage(data=img)

  def toggle_color( self ):
    self.cmode ^= 1
    if( self.cmode==1 ):
      self.cbtn.config(text="Gray")
      self.lframe.pack(side="left", padx=8)
    else:
      self.cbtn.config(text="Color")
      self.lframe.pack_forget()
    self.build_base()
    self.redraw()

  def toggle_range( self ):
    self.rmode ^= 1
    self.rbtn.config(text="Range:16bit" if self.rmode==0 else "Range:Auto")
    self.rebuild_gray()
    self.update_legend()
    self.build_base()
    self.redraw()

  # ---- 画素選択/座標ジャンプ --------------------------------------------------
  def fmt_status( self, ix, iy ):
    i = iy*self.iw+ix
    return " x=%d  y=%d  val=%d (0x%04X)" % (ix, iy, self.vals[i], self.raw[i])

  def center_on( self, ix, iy ):
    s  = self.scale()
    cw = max(self.canvas.winfo_width(), 1)
    ch = max(self.canvas.winfo_height(), 1)
    self.vx = ix+0.5-cw/(2*s)  # 対象画素の中心をキャンバス中央へ
    self.vy = iy+0.5-ch/(2*s)
    self.mark = (ix, iy)
    self.redraw()

  def goto_xy( self ):
    try:
      ix = int(self.xent.get())
      iy = int(self.yent.get())
    except ValueError:
      self.status.config(text=" invalid coordinate")
      return
    if( not(0<=ix<self.iw and 0<=iy<self.ih) ):
      self.status.config(text=" out of range: x=%d y=%d (image %dx%d)"
                         % (ix, iy, self.iw, self.ih))
      return
    self.center_on(ix, iy)
    self.status.config(text=self.fmt_status(ix, iy))

  def select_at( self, cx, cy ):
    s  = self.scale()
    ix = math.floor(self.vx+cx/s)
    iy = math.floor(self.vy+cy/s)
    if( 0<=ix<self.iw and 0<=iy<self.ih ):
      self.mark = (ix, iy)
      self.xent.delete(0, "end")
      self.xent.insert(0, str(ix))
      self.yent.delete(0, "end")
      self.yent.insert(0, str(iy))
      self.redraw()
      self.status.config(text=self.fmt_status(ix, iy))
    else:
      self.clear_mark()  # 画像外クリックで選択解除
      self.status.config(text="")

  def clear_mark( self ):
    self.mark = None
    self.redraw()

  # ---- ズーム/パン ------------------------------------------------------------
  def luma( self, i ):
    if( self.cmode==0 ):
      return self.gray[i]
    return self.lut_luma[self.gray[i]]

  def scale( self ):
    m, d = SCALES[self.si]
    return m/d

  def zoom_text( self ):
    m, d = SCALES[self.si]
    if( d==1 ):
      return "zoom x%d" % m
    return "zoom 1/%d" % d

  def on_configure( self, e ):
    if( not self.init ):
      self.init = True
      s = self.scale()
      self.vx = (self.iw-e.width/s)/2
      self.vy = (self.ih-e.height/s)/2
    self.redraw()

  def on_press( self, e ):
    self.drag  = (e.x, e.y, self.vx, self.vy)
    self.moved = False

  def on_drag( self, e ):
    if( self.drag is None ):
      return
    x0, y0, vx0, vy0 = self.drag
    if( abs(e.x-x0)>3 or abs(e.y-y0)>3 ):
      self.moved = True  # しきい値超えでドラッグ(パン)確定
    if( not self.moved ):
      return
    s = self.scale()
    self.vx = vx0-(e.x-x0)/s
    self.vy = vy0-(e.y-y0)/s
    self.redraw()

  def on_release( self, e ):
    if( self.drag is None ):
      return
    if( not self.moved ):
      self.select_at(self.drag[0], self.drag[1])  # クリック=押した位置の画素を選択
    self.drag = None

  def on_motion( self, e ):
    s  = self.scale()
    ix = math.floor(self.vx+e.x/s)
    iy = math.floor(self.vy+e.y/s)
    if( 0<=ix<self.iw and 0<=iy<self.ih ):
      self.status.config(text=self.fmt_status(ix, iy))
    else:
      self.status.config(text="")

  def on_wheel( self, e ):
    step = 1 if e.delta>0 else -1
    self.zoom_at(e.x, e.y, step)

  def zoom_at( self, cx, cy, step ):
    ni = min(max(self.si+step, 0), len(SCALES)-1)
    if( ni==self.si ):
      return
    s1 = self.scale()
    px = self.vx+cx/s1
    py = self.vy+cy/s1
    self.si = ni
    s2 = self.scale()
    self.vx = px-cx/s2
    self.vy = py-cy/s2
    self.redraw()

  def zoom_step( self, step ):
    cw = max(self.canvas.winfo_width(), 1)
    ch = max(self.canvas.winfo_height(), 1)
    self.zoom_at(cw/2, ch/2, step)

  def zoom_1to1( self ):
    cw = max(self.canvas.winfo_width(), 1)
    ch = max(self.canvas.winfo_height(), 1)
    self.zoom_at(cw/2, ch/2, SCALES.index((1,1))-self.si)

  def zoom_fit( self ):
    cw   = max(self.canvas.winfo_width(), 1)
    ch   = max(self.canvas.winfo_height(), 1)
    best = 0
    for i in range(len(SCALES)):
      m, d = SCALES[i]
      s = m/d
      if( self.iw*s<=cw and self.ih*s<=ch ):
        best = i
    self.si = best
    s = self.scale()
    self.vx = (self.iw-cw/s)/2
    self.vy = (self.ih-ch/s)/2
    self.redraw()

  # ---- 描画 -------------------------------------------------------------------
  def redraw( self ):
    c  = self.canvas
    cw = max(c.winfo_width(), 1)
    ch = max(c.winfo_height(), 1)
    s  = self.scale()
    m, d = SCALES[self.si]
    c.delete("all")
    self.zlabel.config(text=self.zoom_text())

    x0 = max(0, int(self.vx))
    y0 = max(0, int(self.vy))
    x1 = min(self.iw, int(self.vx+cw/s)+1)
    y1 = min(self.ih, int(self.vy+ch/s)+1)
    if( x0>=x1 or y0>=y1 ):
      self.view = None
      return
    if( d>1 ):
      x0 = (x0//d)*d  # subsample位置ずれ防止のアライン
      y0 = (y0//d)*d

    # 可視領域だけをtk側でcrop+zoom/subsample(巨大画像でもメモリ爆発しない)
    # 重要: サイズ指定なしで生成する(width/heightを指定するとcopyで拡張されない)
    self.view = tk.PhotoImage()
    if( d==1 ):
      self.tk.call(str(self.view), "copy", str(self.base),
                   "-from", x0, y0, x1, y1, "-zoom", m, m)
    else:
      self.tk.call(str(self.view), "copy", str(self.base),
                   "-from", x0, y0, x1, y1, "-subsample", d, d)
    c.create_image((x0-self.vx)*s, (y0-self.vy)*s, image=self.view, anchor="nw")

    if( s>=VAL_MIN_SCALE ):
      fs = max(6, int(s)//8)
      for iy in range(y0, y1):
        for ix in range(x0, x1):
          px = (ix-self.vx)*s
          py = (iy-self.vy)*s
          c.create_rectangle(px, py, px+s, py+s, outline="#808080")
          i  = iy*self.iw+ix
          fg = "white" if self.luma(i)<128 else "black"
          if( s>=COORD_MIN_SCALE ):
            c.create_text(px+s/2, py+s/2-fs, text="%d,%d" % (ix, iy),
                          fill=fg, font=("TkDefaultFont", fs))
            c.create_text(px+s/2, py+s/2+fs, text=str(self.vals[i]),
                          fill=fg, font=("TkDefaultFont", fs, "bold"))
          else:
            c.create_text(px+s/2, py+s/2, text=str(self.vals[i]),
                          fill=fg, font=("TkDefaultFont", fs))

    # 選択中画素のマーカー(クリック/Go/PrevDiff/NextDiffで設定、Escで解除)
    if( self.mark is not None ):
      mx, my = self.mark
      px0 = (mx-self.vx)*s
      py0 = (my-self.vy)*s
      pcx = (mx+0.5-self.vx)*s
      pcy = (my+0.5-self.vy)*s
      c.create_rectangle(px0, py0, px0+s, py0+s, outline="red", width=2)
      if( s<24 ):
        # セルが小さく赤枠だけでは見えないときのみ固定サイズの照準を重ねる
        r = 12
        c.create_line(pcx-r, pcy, pcx+r, pcy, fill="red")
        c.create_line(pcx, pcy-r, pcx, pcy+r, fill="red")
        c.create_oval(pcx-r, pcy-r, pcx+r, pcy+r, outline="red")

    # 他ウィンドウへ表示状態を同期
    if( self.sync is not None ):
      self.sync.push(self)
