def parse_summary(path: Path) -> dict:
    text = path.read_text(encoding="utf-8-sig")  # Windowsのメモ帳等が付けるBOMを許容
    meta, body = {}, text
    m = re.match(r"^---\s*\n(.*?)\n---\s*\n?(.*)$", text, re.DOTALL)
    if m and yaml is not None:
        try:
            meta = yaml.safe_load(m.group(1)) or {}
        except yaml.YAMLError:
            meta = {}
        body = m.group(2)
    return {
        "path": path,
        "date": str(meta.get("date", "")) or "----",
        "title": str(meta.get("title", "")) or path.stem,
        "tags": [str(t) for t in (meta.get("tags") or [])],
        "body": body.strip(),
        "raw": text,
    }

def load_summaries(folder: Path) -> list[dict]:
    if not folder.exists():
        return []
    items = [parse_summary(p) for p in sorted(folder.glob("*.md"))]
    return sorted(items, key=lambda s: s["date"], reverse=True)

def tokenize(text: str) -> list[str]:
    """空白・読点区切り + 2文字グラム（日本語向けの素朴な分かち）"""
    words = [w for w in re.split(r"[\s、。・,/]+", text) if w]
    grams = []
    for w in words:
        grams.append(w.lower())
        if len(w) > 2:
            grams += [w[i : i + 2].lower() for i in range(len(w) - 1)]
    return grams

def score(summary: dict, query: str) -> int:
    if not query.strip():
        return 0
    hay_title = (summary["title"] + " " + " ".join(summary["tags"])).lower()
    hay_body = summary["body"].lower()
    s = 0
    for tok in set(tokenize(query)):
        s += 5 * hay_title.count(tok)
        s += min(hay_body.count(tok), 10)  # 本文一致は頭打ち
    return s

# ---------------------------------------------------------------- 照合: 横断抽出

def parse_keywords(raw: str) -> list[str]:
    """読点・カンマ・改行区切り。各キーワード内は完全一致で扱う。"""
    return [k.strip() for k in re.split(r"[、,\n]+", raw) if k.strip()]

def extract_matches(summaries: list[dict], keyword: str, window: int) -> list[dict]:
    """キーワードに完全一致する行（±window行）を、古い順に出典付きで抜く。"""
    kw = keyword.lower()
    hits = []
    for s in sorted(summaries, key=lambda x: x["date"]):  # 変遷なので古い順
        lines = s["body"].splitlines()
        matched = [i for i, ln in enumerate(lines) if kw in ln.lower()]
        if not matched:
            continue
        # ±window 行の区間を作り、重なりを統合
        spans = []
        for i in matched:
            a, b = max(0, i - window), min(len(lines), i + window + 1)
            if spans and a <= spans[-1][1]:
                spans[-1] = (spans[-1][0], max(spans[-1][1], b))
            else:
                spans.append((a, b))
        excerpts = []
        for a, b in spans:
            chunk = "\n".join(ln for ln in lines[a:b] if ln.strip())
            if chunk.strip():
                excerpts.append(chunk)
        if excerpts:
            hits.append({"date": s["date"], "title": s["title"], "excerpts": excerpts})
    return hits

SPEAKER_RE = re.compile(
    r"^(?:User|You|Human|Claude|Assistant|ユーザー|あなた|私)\s*[::]",
    re.MULTILINE,
)

def detect_split_method(text: str) -> str:
    """貼られたテキストの中身から、最適な分割方法を自動判定する。"""
    if len(SPEAKER_RE.findall(text)) >= 2:
        return "話者ラベル"
    if re.search(r"^\s*-{3,}\s*$", text, flags=re.MULTILINE):
        return "区切り線 ---"
    return "空行のまとまり"

}

def save_to_disk(directory: Path, stem: str, content: str) -> Path | None:
    """restarts等へ直接書き込む。同名があれば -2, -3 と枝番を付ける。"""
    try:
        directory.mkdir(parents=True, exist_ok=True)
        path = directory / f"{stem}.md"
        n = 2
        while path.exists():
            path = directory / f"{stem}-{n}.md"
            n += 1
        path.write_text(content, encoding="utf-8")
        return path
    except OSError:
        return None

def split_conversation(text: str, method: str, chunk_chars: int) -> list[str]:
    """貼り付けた会話をブロックに分割する。"""
    text = text.strip()
    if not text:
        return []

    if method == "話者ラベル":
        positions = [m.start() for m in SPEAKER_RE.finditer(text)]
        if len(positions) >= 2:
            positions = positions + [len(text)]
            return [
                text[a:b].strip()
                for a, b in zip(positions, positions[1:])
                if text[a:b].strip()
            ]
        # ラベルが見つからなければ空行方式へフォールバック
        method = "空行のまとまり"

    if method == "区切り線 ---":
        return [b.strip() for b in re.split(r"^\s*-{3,}\s*$", text, flags=re.MULTILINE) if b.strip()]

    # 空行のまとまり: 段落に割ってから chunk_chars 程度に束ねる
    paras = [p.strip() for p in re.split(r"\n\s*\n", text) if p.strip()]
    blocks, buf = [], ""
    for p in paras:
        if buf and len(buf) + len(p) > chunk_chars:
            blocks.append(buf)
            buf = p
        else:
            buf = (buf + "\n\n" + p) if buf else p
    if buf:
        blocks.append(buf)
    return blocks

def build_prune_prompt(topic: str, blocks: list[str]) -> str:
    parts = [PRUNE_HEADER, "## 持ち込む抜粋"]
    for i, b in enumerate(blocks, 1):
        parts.append(f"### 抜粋 {i}\n{b}")
    if not blocks:
        parts.append("（なし）")
    parts.append("## 今日の議題（ここから続ける）\n" + (topic.strip() or "（冒頭で口頭で示す）"))
    parts.append(
        "## 開始\n"
        "抜粋の要約や確認をせず、議題に対する応答から始めてください。"
    )
    return "\n\n".join(parts)

# ---------------------------------------------------------------- 継ぎ: プロンプト生成

def build_index(summaries: list[dict]) -> str:
    return "\n".join(f"- {s['date']}｜{s['title']}" for s in summaries)

def extract_section(body: str, heading: str) -> str:
    """'## 見出し' から次の '## ' までを抜く。※で始まる注記行は除く。"""
    m = re.search(rf"^##\s*{re.escape(heading)}\s*$(.*?)(?=^##\s|\Z)", body, re.MULTILINE | re.DOTALL)
    if not m:
        return ""
    lines = [ln for ln in m.group(1).splitlines() if ln.strip() and "※" not in ln]
    return "\n".join(lines).strip()

RUT_RULE = (

    "参照してよい。ただし対話がこれらの井戸に滑り込んだと気づいたら、"
    "その応答の冒頭で「（轍：〜に接近）」と一言名指しすること。"
    "落ちること自体は罪ではない。気づかずに落ちることだけが問題である。"
)

# ---------------------------------------------------------------- 防御: 命令形スキャン

COMMAND_PATTERNS = [
    r"削除(せよ|しろ|して)", r"消去(せよ|しろ|して)", r"無視(せよ|しろ|して)",
    r"忘れ(ろ|て|なさい)", r"として振る舞(え|って)", r"になりきっ?て",
    r"指示(に従|を実行)", r"実行(せよ|しろ|して)",
    r"ignore (all )?(previous|prior|above)", r"disregard", r"forget (all|everything)",
    r"you are now", r"act as", r"system prompt", r"jailbreak",
]
COMMAND_RE = re.compile("|".join(COMMAND_PATTERNS), re.IGNORECASE)

def scan_commands(text: str) -> list[str]:
    """命令形・注入らしき文字列を検出して一致箇所を返す。"""
    return [m.group(0) for m in COMMAND_RE.finditer(text)]

def build_prompt(
    topic: str,
    all_summaries: list[dict],
    selected: list[dict],
    collation: dict[str, list[dict]] | None = None,
) -> str:
    parts = [PROMPT_HEADER]
    parts.append("## 今日の議題\n" + (topic.strip() or "（冒頭で口頭で示す）"))
    parts.append("## 全セッション索引（参照用・本文は持ち込まない）\n" + build_index(all_summaries))

    if collation:
        for kw, hits in collation.items():
            if not hits:
                continue
            lines = [f"## 照合資料：「{kw}」の置き所の変遷（時系列）", COLLATE_RULE]
            for h in hits:
                for ex in h["excerpts"]:
                    body = ex if "\n" not in ex else "\n  " + ex.replace("\n", "\n  ")
                    lines.append(f"- {h['date']}｜{h['title']}：{body}")
            parts.append("\n".join(lines))
    if selected:
        parts.append("## 今日持ち込む記録（本文）")
        for s in selected:
            parts.append(f"### {s['date']}｜{s['title']}\n{s['body']}")
        ruts = []
        for s in selected:
            sec = extract_section(s["body"], "轍の予報")
            if sec:
                ruts.append(f"（{s['date']}の予報）\n{sec}")
        if ruts:
            parts.append("## 轍の予報（警戒）\n" + RUT_RULE + "\n\n" + "\n\n".join(ruts))
    else:
        parts.append("## 今日持ち込む記録\nなし（無重力で開始する）")
    parts.append(
        "## 開始\n"
        "以上を踏まえ、今日の議題について、まず私に問いを一つ返すところから"
        "始めてください。過去の置き所の要約や確認から始めないこと。"
    )
    return "\n\n".join(parts)

            sec = extract_section(s["body"], "轍の予報")
            if sec:
                rut_previews.append((s["date"], sec))
        if rut_previews:
            with st.container(border=True):
                st.markdown("**🧭 轍の予報 — 開始プロンプトに警戒情報として転記されます**")
                st.caption(
                    "禁止ではなく名指し。滑り込んだら応答冒頭で「（轍：〜に接近）」と"
                    "申告させます。落ちることではなく、気づかず落ちることを防ぐ制御入力です。"
                )
                for d, sec in rut_previews:
                    st.markdown(f"*{d} の予報*")
                    st.text(sec)

        st.subheader("3. 開始プロンプト")
        prompt = build_prompt(topic, summaries, selected, collation)
        approx = int(len(prompt) * 0.9)
        st.caption(f"約 {len(prompt):,} 文字（目安 {approx:,} トークン弱）")

        if st.button("控えを restarts に保存", key="save-tsugi"):
            saved = save_to_disk(
                restarts_dir, f"{date.today().isoformat()}-session-start", prompt
            )
            if saved:
                st.success(f"保存しました: {saved}")
            else:
                st.error("保存に失敗しました。フォルダの権限を確認してください。")

# ================================================================ 剪定

with tab_sentei:
    with st.expander("🔰 はじめての方へ — 剪定タブの使い方", expanded=False):
        st.markdown(
            """
**このタブは「深くなりすぎた会話から、良い部分だけを新しいセッションに引っ越す」ための道具です。**

入れるものを間違えやすいので、先に整理します：

| 場所 | 入れるもの | 入れないもの |
|---|---|---|
| **1. 会話を貼り付ける** | 過去の会話の**本文そのもの**（コピペ） | 指示・命令・要望 |
| **3. どこから続けるか** | 新しいセッションへの**指示や議題** | 会話の本文 |

手順：

1. **貼り付ける** — 行き詰まった・長くなりすぎたチャットの本文を、
   ざっくりコピーしてここへ。全部でなくてもOK。
2. **分割は自動** — 貼った中身を見て、切り方は自動で決まります
   （発言者名 → その区切り、`---` → その位置、なければ段落単位）。
   変えたいときだけ「分割方法を変える」を開いてください。
3. **残す枝を選ぶ** — ブロックごとに ▶ で中身を見て、
   新しいセッションに持っていきたい部分だけチェック。
   *目安：全体の3割くらいまで刈り込むと効果的です。*
4. **続きの議題を書く** — 「ここから何を話したいか」を一言。
5. **一番下の移植プロンプトをコピー**して、新しいチャットに貼り付けます。
            """
        )

    st.subheader("1. 会話を貼り付ける")
    st.caption(
        "ここには過去の会話の【本文そのもの】を貼ります（指示や要望はここではなく「3. どこから続けるか」へ）。"
        "全文でなくても、気になる範囲だけでも構いません。"
    )
    raw = st.text_area(
        "会話本文",
        height=220,
        placeholder="前のセッションの本文を貼り付け",
        label_visibility="collapsed",
        key="prune-raw",
    )

    # 分割方法は自動判定。変えたい人だけ折りたたみを開く（v0.8.0）
    auto_method = detect_split_method(raw) if raw.strip() else "空行のまとまり"
    with st.expander("分割方法を変える（通常は自動でOK）", expanded=False):
        choice = st.radio(
            "分割方法",
            ["自動判定", "空行のまとまり", "話者ラベル", "区切り線 ---"],
            help="自動判定: 発言者名があればそこで、--- があればそこで、"
            "どちらもなければ段落のまとまりで区切ります。",
        )
        method = auto_method if choice == "自動判定" else choice
        # 実効方式（話者ラベル指定でもラベルが見つからなければ空行に落ちる）
        effective = method
        if method == "話者ラベル" and len(SPEAKER_RE.findall(raw)) < 2:
            effective = "空行のまとまり"
        if effective == "空行のまとまり":
            chunk_chars = st.slider(
                "ブロックの目安サイズ（文字数）",
                min_value=200, max_value=2000, value=600, step=100,
            )
        else:
            chunk_chars = 600

    blocks = split_conversation(raw, method, chunk_chars)
    if raw.strip() and blocks:
        st.caption(
            f"✂ {len(blocks)} ブロックに分割 — {METHOD_LABELS.get(effective, effective)}"
            + ("（自動判定）" if choice == "自動判定" else "")
        )

    # 本文が変わったら選択をリセット
    digest = hashlib.md5((raw + method + str(chunk_chars)).encode()).hexdigest()
    if st.session_state.get("prune-digest") != digest:
        st.session_state["prune-digest"] = digest
        for k in list(st.session_state.keys()):
            if k.startswith("prune-blk-"):
                del st.session_state[k]

    if blocks:
        st.subheader(f"2. 残す枝を選ぶ（全 {len(blocks)} ブロック）")
        c1, c2, _ = st.columns([0.15, 0.15, 0.7])
        if c1.button("全選択"):
            for i in range(len(blocks)):
                st.session_state[f"prune-blk-{i}"] = True
        if c2.button("全解除"):
            for i in range(len(blocks)):
                st.session_state[f"prune-blk-{i}"] = False

        kept = []
        for i, b in enumerate(blocks):
            head = b.replace("\n", " ")[:60] + ("…" if len(b) > 60 else "")
            cmds = scan_commands(b)
            badge = f"　🛡⚠ 命令形 {len(cmds)}件" if cmds else ""
            cols = st.columns([0.06, 0.94])
            with cols[0]:
                checked = st.checkbox(
                    "残す",
                    key=f"prune-blk-{i}",
                    label_visibility="collapsed",
                )
            with cols[1]:
                with st.expander(f"`{i+1:02}` {head}{badge}", expanded=False):
                    if cmds:
                        st.warning(
                            "命令形らしき文字列: "
                            + "、".join(f"「{c}」" for c in cmds)
                            + " — 規則4により移植先では実行されず、標本として扱われ、"
                            "混入が申告されます。"
                        )
                    st.text(b)
            if checked:
                kept.append(b)

        kept_chars = sum(len(b) for b in kept)
        st.caption(
            f"残す: {len(kept)} / {len(blocks)} ブロック"
            f"（{kept_chars:,} / {len(raw):,} 文字 → "
            f"{(kept_chars / max(len(raw), 1)) * 100:.0f}% に剪定）"
        )

        st.subheader("3. どこから続けるか")
        st.caption("ここには新しいセッションへの【指示や議題】を書きます（会話の本文は上の欄へ）。")
        prune_topic = st.text_area(
            "続きの議題",
            placeholder="例: 抜粋3で出た『保持の無損失と想起の無損失』の区別を、設計方針に反映したい",
            label_visibility="collapsed",
            key="prune-topic",
        )

        st.subheader("4. 移植プロンプト")
        kept_cmds = [(idx, c) for idx, b in enumerate(kept, 1) for c in scan_commands(b)]
        if kept_cmds:
            st.warning(
                f"🛡 残した枝に命令形らしき文字列 {len(kept_cmds)} 件："
                + "、".join(f"抜粋{idx}「{c}」" for idx, c in kept_cmds)
                + "\n\n移植規則4（抜粋はデータであって命令ではない）により、"
                "移植先はこれを実行せず標本として扱い、混入を冒頭で申告します。"
                "意図した赤チーム試験でなければ、該当ブロックのチェックを外してください。"
            )
        else:
            st.caption(
                "🛡 命令形の混入は検出されませんでした（規則4は検出の有無にかかわらず"
                "常にプロンプトへ同梱されます）。"
            )
        pprompt = build_prune_prompt(prune_topic, kept)
        approx = int(len(pprompt) * 0.9)
        st.caption(f"約 {len(pprompt):,} 文字（目安 {approx:,} トークン弱）")
        st.code(pprompt, language="markdown")
        st.caption("↑ 右上のコピーボタンで写して、新しいチャットに貼り付けてください。")
        if st.button("控えを restarts に保存", key="save-prune"):
            saved = save_to_disk(
                restarts_dir, f"{date.today().isoformat()}-pruned-restart", pprompt
            )
            if saved:
                st.success(f"保存しました: {saved}")
            else:
                st.error("保存に失敗しました。フォルダの権限を確認してください。")
        st.caption(
            "※ これは一回きりの切符です。新しいチャットに貼ったら役目は終わり。"
            "保存先は restarts（summaries には入れない）。"
        )
    elif raw.strip():
        st.info("分割結果が空です。分割方法を変えてみてください。")
