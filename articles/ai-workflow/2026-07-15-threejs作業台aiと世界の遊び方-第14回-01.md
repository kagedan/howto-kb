<script>
(function() {

const canvas = document.getElementById('eb-canvas');
const wrap = document.getElementById('eb-wrap');
const renderer = new THREE.WebGLRenderer({ canvas, antialias: true });
renderer.setPixelRatio(Math.min(devicePixelRatio, 2));
renderer.shadowMap.enabled = true;

const scene = new THREE.Scene();
scene.background = new THREE.Color(0xe4e8ec);
const camera = new THREE.PerspectiveCamera(50, 1, 0.1, 200);

function resize() {
  const w = wrap.clientWidth, h = wrap.clientHeight;
  renderer.setSize(w, h, false);
  camera.aspect = w / h;
  camera.updateProjectionMatrix();
}

scene.add(new THREE.AmbientLight(0xffffff, 0.3));
const sun = new THREE.DirectionalLight(0xfff8f0, 0.6); sun.position.set(5,8,4); sun.castShadow=true; scene.add(sun);
const fill = new THREE.DirectionalLight(0xd0e0ff, 0.4); fill.position.set(-4,2,-3); scene.add(fill);

const ground = new THREE.Mesh(new THREE.PlaneGeometry(40,40), new THREE.MeshPhongMaterial({color:0xc8ccc8}));
ground.rotation.x=-Math.PI/2; ground.position.y=-5; ground.receiveShadow=true; scene.add(ground);

const allMeshes=[];

const GEO_NAMES={
  IcosahedronGeometry:'二十面体', OctahedronGeometry:'八面体',
  DodecahedronGeometry:'正十二面体', TetrahedronGeometry:'四面体',
  TorusGeometry:'トーラス', TorusKnotGeometry:'トーラスノット',
  ConeGeometry:'円錐', CylinderGeometry:'円柱',
  SphereGeometry:'球', BoxGeometry:'直方体',
  PlaneGeometry:'平面', BufferGeometry:'カスタム',
};
function getLabel(m){
  if(m.name) return m.name;
  const base=GEO_NAMES[m.geometry.type]||m.geometry.type.replace('Geometry','');
  return base+(m.userData.label?' '+m.userData.label:'');
}
const hudList=document.getElementById('hud-list');
let _hudCount=-1;
function updateHUD(){
  _hudCount=allMeshes.length;
  hudList.innerHTML=allMeshes.length
    ?allMeshes.map(m=>`· ${getLabel(m)}`).join('<br>')
    :'<span style="color:#ccc">（なし）</span>';
}

const posV=[],negV=[];
for(let i=-10;i<=10;i++){const arr=i<0?negV:posV;arr.push(i,0,0);arr.push(0,i,0);arr.push(0,0,i);}
const axPosGeo=new THREE.BufferGeometry();
axPosGeo.setAttribute('position',new THREE.Float32BufferAttribute(posV,3));
scene.add(new THREE.Points(axPosGeo,new THREE.PointsMaterial({color:0x000000,size:4,sizeAttenuation:false})));
const axNegGeo=new THREE.BufferGeometry();
axNegGeo.setAttribute('position',new THREE.Float32BufferAttribute(negV,3));
scene.add(new THREE.Points(axNegGeo,new THREE.PointsMaterial({color:0x5070D0,size:4,sizeAttenuation:false})));

let dragging=false,prevX=0,prevY=0,theta=0.5,phi=0.3,R_val=22.0;
const tgt=new THREE.Vector3(0,0,0);
const updateCam=()=>{
  camera.position.set(tgt.x+R_val*Math.sin(theta)*Math.cos(phi),tgt.y+R_val*Math.sin(phi),tgt.z+R_val*Math.cos(theta)*Math.cos(phi));
  camera.lookAt(tgt);
};
updateCam();

canvas.addEventListener("mousedown",e=>{dragging=true;prevX=e.clientX;prevY=e.clientY;});
window.addEventListener("mouseup",()=>dragging=false);
window.addEventListener("mousemove",e=>{if(!dragging)return;theta-=(e.clientX-prevX)*0.007;phi=Math.max(0.05,Math.min(1.2,phi+(e.clientY-prevY)*0.007));prevX=e.clientX;prevY=e.clientY;updateCam();});
canvas.addEventListener("touchstart",e=>{dragging=true;prevX=e.touches[0].clientX;prevY=e.touches[0].clientY;},{passive:true});
window.addEventListener("touchend",()=>dragging=false);
window.addEventListener("touchmove",e=>{if(!dragging)return;theta-=(e.touches[0].clientX-prevX)*0.007;phi=Math.max(0.05,Math.min(1.2,phi+(e.touches[0].clientY-prevY)*0.007));prevX=e.touches[0].clientX;prevY=e.touches[0].clientY;updateCam();},{passive:true});
canvas.addEventListener("wheel",e=>{R_val=Math.max(6,Math.min(60,R_val+e.deltaY*0.02));updateCam();},{passive:true});

resize();
window.addEventListener("resize",resize);
(function animate(){requestAnimationFrame(animate);renderer.render(scene,camera);if(allMeshes.length!==_hudCount)updateHUD();})();

})();
</script>
