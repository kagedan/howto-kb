いちいち切り替えたり起動したりがめんどくて手元でやりたいときにあると便利

```
function howto() {
  local backend="claude"
  while [[ "$1" == -* ]]; do
    case "$1" in
      -c|--codex)  backend="codex";  shift ;;
      -g|--gemini) backend="gemini"; shift ;;
      --claude)    backend="claude"; shift ;;
      --) shift; break ;;
      *) break ;;
    esac
  done

  local text="${*:-$(cat)}"
  local result
  local prompt="Output ONLY a single shell one-liner for ${OSTYPE} (${SHELL:t}) that accomplishes the task. No explanations, no markdown, no code fences, no comments, no trailing newline. Just the raw command. Task:"

  case "$backend" in
    claude)
      result=$(printf '%s' "$text" | claude -p --model sonnet --no-session-persistence "$prompt" 2>&1)
      ;;
    codex)
      result=$(codex exec --model gpt-5-mini --color never "${prompt} ${text}" 2>&1)
      ;;
    gemini)
      result=$(printf '%s' "$text" | gemini -m gemini-2.5-flash -p "$prompt" 2>&1)
      ;;
  esac

  printf '%s' "$result" | pbcopy
  printf '%s\n' "$result"
}
```

使い方

```
howto "カレントディレクトリ以下の .log を再帰削除" # default claude
howto -c "直近5コミットを作者別に集計"           # codex
howto -g "gitで変更行数の多いファイル top10"     # gemini
echo "pod が Pending のものだけ取得" | howto
```
