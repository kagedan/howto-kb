"""
cluster.py
----------
統合ベクトル(combined)を使った教師なしクラスタリング(自動グループ化)。

手順:
  1. vectors.npz の combined ベクトルを読む
  2. k を範囲探索し、シルエット係数が最大の k を採用(KMeans)
  3. 各クラスタを「頻出キーワード + 代表タイトル」で自動命名
  4. クラスタ割り当てを ChromaDB の各コレクション metadata に書き戻す
  5. クラスタ重心(centroid)を保存 -> 新規文書の最近傍クラスタ分類に使う
  6. レポート(JSON / Markdown)を出力

KMeans を既定にする理由:
  - 正規化済みベクトル上の KMeans は球面 k-means 相当でコサイン類似に整合
  - 全件にラベルが付く(HDBSCAN はノイズ点が残る)
  HDBSCAN を使いたい場合は --algo hdbscan(密度ベース, 自動クラスタ数)。
"""
from __future__ import annotations
import os, json, argparse
from collections import Counter
import numpy as np

def load_vectors(index_dir):
    z = np.load(os.path.join(index_dir, "vectors.npz"), allow_pickle=True)
    return list(z["ids"]), z["combined"]

def load_metas(index_dir):
    import chromadb
    client = chromadb.PersistentClient(path=os.path.join(index_dir, "chroma_db"))
    col = client.get_collection("docs_combined")
    got = col.get(include=["metadatas"])
    return dict(zip(got["ids"], got["metadatas"]))

def choose_k_kmeans(X, kmin=8, kmax=40, sample=4000, seed=42):
    from sklearn.cluster import KMeans
    from sklearn.metrics import silhouette_score
    rng = np.random.default_rng(seed)
    idx = rng.choice(len(X), size=min(sample, len(X)), replace=False)
    Xs = X[idx]
    best = (-1, None, None)
    history = []
    for k in range(kmin, kmax + 1, 2):
        km = KMeans(n_clusters=k, random_state=seed, n_init=4).fit(Xs)
        s = silhouette_score(Xs, km.labels_, metric="cosine")
        history.append((k, float(s)))
        print(f"  k={k:>3}  silhouette={s:.4f}")
        if s > best[0]:
            best = (s, k, None)
    return best[1], best[0], history

def name_clusters(labels, ids, metas, topn_kw=5, topn_title=3):
    clusters = {}
    by_label = {}
    for lab, _id in zip(labels, ids):
        by_label.setdefault(int(lab), []).append(_id)
    for lab, members in sorted(by_label.items()):
        kwc = Counter()
        titles = []
        for _id in members:
            m = metas.get(_id, {})
            for k in str(m.get("keywords", "")).split():
                kwc[k] += 1
            if m.get("title"):
                titles.append(m["title"])
        top_kw = [w for w, _ in kwc.most_common(topn_kw)]
        label_name = " / ".join(top_kw) if top_kw else f"cluster_{lab}"
        clusters[lab] = {
            "cluster_id": lab,
            "size": len(members),
            "label": label_name,
            "top_keywords": top_kw,
            "sample_titles": titles[:topn_title],
            "member_ids": members,
        }
    return clusters

def run(index_dir, algo="kmeans", kmin=8, kmax=40, fixed_k=None, seed=42):
    ids, X = load_vectors(index_dir)
    metas = load_metas(index_dir)
    X = np.asarray(X, dtype=np.float32)
    print(f"[cluster] {len(ids)} vectors dim={X.shape[1]} algo={algo}")

    centroids = None
    if algo == "kmeans":
        from sklearn.cluster import KMeans
        if fixed_k:
            k, sil = fixed_k, None
            print(f"[cluster] fixed k={k}")
        else:
            k, sil, hist = choose_k_kmeans(X, kmin, kmax, seed=seed)
            print(f"[cluster] selected k={k} (silhouette={sil:.4f})")
        km = KMeans(n_clusters=k, random_state=seed, n_init=8).fit(X)
        labels = km.labels_
        # 正規化重心(コサイン最近傍分類用)
        centroids = km.cluster_centers_
        centroids = centroids / np.clip(np.linalg.norm(centroids, axis=1, keepdims=True), 1e-9, None)
    elif algo == "hdbscan":
        import hdbscan
        cl = hdbscan.HDBSCAN(min_cluster_size=15, metric="euclidean")
        labels = cl.fit_predict(X)  # -1 はノイズ
        uniq = [c for c in sorted(set(labels)) if c != -1]
        centroids = np.vstack([X[labels == c].mean(0) for c in uniq])
        centroids = centroids / np.clip(np.linalg.norm(centroids, axis=1, keepdims=True), 1e-9, None)
        print(f"[cluster] hdbscan -> {len(uniq)} clusters, noise={int((labels==-1).sum())}")
    else:
        raise ValueError(algo)

    clusters = name_clusters(labels, ids, metas, )

    # ---- ChromaDB に cluster_id / cluster_label を書き戻す ----
    import chromadb
    client = chromadb.PersistentClient(path=os.path.join(index_dir, "chroma_db"))
    id2lab = {i: int(l) for i, l in zip(ids, labels)}
    for cname in ("docs_summary", "docs_keywords", "docs_combined"):
        col = client.get_collection(cname)
        got = col.get(include=["metadatas"])
        new_ids, new_meta = [], []
        for _id, m in zip(got["ids"], got["metadatas"]):
            lab = id2lab.get(_id, -1)
            m = dict(m)
            m["cluster_id"] = lab
            m["cluster_label"] = clusters.get(lab, {}).get("label", "noise")
            new_ids.append(_id); new_meta.append(m)
        for i in range(0, len(new_ids), 256):
            col.update(ids=new_ids[i:i+256], metadatas=new_meta[i:i+256])
    print("[cluster] wrote cluster_id back to ChromaDB metadata")

    # ---- centroids 保存 ----
    cl_ids = sorted(clusters.keys())
    np.savez_compressed(os.path.join(index_dir, "centroids.npz"),
                        cluster_ids=np.array(cl_ids),
                        centroids=np.asarray(centroids, dtype=np.float32),
                        labels=np.array([clusters[c]["label"] for c in cl_ids], dtype=object))

    # ---- レポート ----
    report = {"algo": algo, "n_docs": len(ids), "n_clusters": len(clusters),
              "clusters": [{k: v for k, v in c.items() if k != "member_ids"}
                           for c in clusters.values()]}
    json.dump(report, open(os.path.join(index_dir, "cluster_report.json"), "w"),
              ensure_ascii=False, indent=2)

    md = ["# クラスタリング結果レポート\n",
          f"- アルゴリズム: **{algo}**",
          f"- 文書数: **{len(ids)}**",
          f"- クラスタ数: **{len(clusters)}**\n",
          "| ID | 件数 | ラベル(頻出キーワード) | 代表タイトル例 |",
          "|---:|---:|---|---|"]
    for c in sorted(clusters.values(), key=lambda x: -x["size"]):
        st = " ; ".join(t[:28] for t in c["sample_titles"])
        md.append(f"| {c['cluster_id']} | {c['size']} | {c['label']} | {st} |")
    open(os.path.join(index_dir, "cluster_report.md"), "w").write("\n".join(md))
    print(f"[cluster] report -> cluster_report.json / cluster_report.md")
    return report

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--index-dir", required=True)
    ap.add_argument("--algo", default="kmeans", choices=["kmeans", "hdbscan"])
    ap.add_argument("--kmin", type=int, default=8)
    ap.add_argument("--kmax", type=int, default=40)
    ap.add_argument("--fixed-k", type=int, default=None)
    a = ap.parse_args()
    run(a.index_dir, a.algo, a.kmin, a.kmax, a.fixed_k)
