### 5.AIと一緒にカレンダーを作ろう！💻🚀

さあ、ここがこの記事の中で最も重要な部分です。

「百聞は一見に如かず」。実際にAIを使って、Excelにカレンダーを作成するマクロ（VBAコード）を作ってみましょう。

あなたは、Excelを立ち上げて、AIに指示を出す準備をするだけです。

【手順1】AIへの指示（プロンプト）の作成

まずは、AI（ChatGPT、Gemini、Claudeなど）に、「どんなコードを作ってほしいか」を日本語で伝えます。この指示文を「プロンプト」と呼びます。

![](https://assets.st-note.com/img/1776556170-uj6ZqxGY7yOCEP1TUhtvHrfg.png)

ここが一番楽しい作業でもあります

コードを作成するのは現在のところ「Claude」が一番優秀だと思います。他のAIでも不具合を指示してあげれば、最終的には動作するものが完成します。

今回は、入力情報にある要件を満たすために、こんなプロンプトを作成して、AIにお願いしてみます。

> ExcelでInputBoxで指定した期間（開始年月〜終了年月）のカレンダーを自動作成するVBAコードを作成してください。  
> 1. InputBoxを2回出し、『開始年月(YYYY/MM)』と『終了年月(YYYY/MM)』を入力させること。  
> 2. 入力された期間の各月について、新しいシートを1枚ずつ作成すること（シート名は『2024年10月』形式）。  
> 3. 各シートの上部に月、その下に日〜土の曜日ヘッダーを付け、その下にカレンダーを配置すること。   
> 4. 土曜日を青、日曜日を赤、国民の祝日を赤にするロジックを含めること（祝日リストはVBAコード内で定数として持つか、AIで生成して）。  
> 5. 作成されたシートは見やすく整形すること。エラーハンドリング（不正な日付入力、同名シート存在時）も入れること。コードはコメントを充実させ、分かりやすくしてください。

ここまで書けば完璧ですが、ニュアンスだけでも充分伝わるので、もっと簡略でも大丈夫です。

【手順2】AIからの回答とコードのコピー

上記プロンプトをAIに送信すると、AIは驚くべき速さでVBAコードを生成してくれます。

例えば、以下のようなコードが返ってくるはずです（これはあくまで一例です）。Claudeで作成しています。

'==============================================================================
' モジュール名: CalendarMaker
' 概要: 指定した期間（開始年月〜終了年月）のカレンダーシートを自動生成するマクロ
' 使用方法: CalendarMaker マクロを実行し、InputBox に開始年月・終了年月を入力
'==============================================================================

Option Explicit

'------------------------------------------------------------------------------
' 祝日リスト（2024〜2026年度 日本の国民の祝日）

'==============================================================================
' メインマクロ: CalendarMaker
' ・InputBox で開始年月・終了年月を取得
' ・各月のカレンダーシートを生成する
'==============================================================================
Sub CalendarMaker()

    Dim startInput  As String   ' 開始年月入力文字列
    Dim endInput    As String   ' 終了年月入力文字列
    Dim startDate   As Date     ' 開始月の1日
    Dim endDate     As Date     ' 終了月の1日
    Dim currentDate As Date     ' ループ用カレント月
    Dim holidays    As Object   ' 祝日辞書

    '--------------------------------------------------------------------------
    ' ① 開始年月の入力
    '--------------------------------------------------------------------------
    Do
        startInput = InputBox( _
            "開始年月を入力してください（例: 2024/10）", _
            "カレンダー作成 - 開始年月")

        ' キャンセルまたは空欄 → マクロ終了
        If startInput = "" Then
            MsgBox "キャンセルされました。", vbInformation, "カレンダー作成"
            Exit Sub
        End If

        ' 入力値の検証
        If IsValidYearMonth(startInput) Then
            startDate = DateSerial(CInt(Left(startInput, 4)), _
                                   CInt(Mid(startInput, 6, 2)), 1)
            Exit Do
        Else
            MsgBox "入力形式が正しくありません。YYYY/MM 形式で入力してください。" & vbCrLf & _
                   "例: 2024/10", vbExclamation, "入力エラー"
        End If
    Loop

    '--------------------------------------------------------------------------
    ' ② 終了年月の入力
    '--------------------------------------------------------------------------
    Do
        endInput = InputBox( _
            "終了年月を入力してください（例: 2025/03）", _
            "カレンダー作成 - 終了年月")

        If endInput = "" Then
            MsgBox "キャンセルされました。", vbInformation, "カレンダー作成"
            Exit Sub
        End If

        If IsValidYearMonth(endInput) Then
            endDate = DateSerial(CInt(Left(endInput, 4)), _
                                 CInt(Mid(endInput, 6, 2)), 1)

            ' 終了月が開始月より前になっていないかチェック
            If endDate < startDate Then
                MsgBox "終了年月は開始年月以降を指定してください。", _
                       vbExclamation, "入力エラー"
            Else
                Exit Do
            End If
        Else
            MsgBox "入力形式が正しくありません。YYYY/MM 形式で入力してください。" & vbCrLf & _
                   "例: 2025/03", vbExclamation, "入力エラー"
        End If
    Loop

    '--------------------------------------------------------------------------
    ' ③ 祝日辞書の取得・シート生成ループ
    '--------------------------------------------------------------------------
    Set holidays = GetHolidays()

    ' 画面更新を止めて高速化
    Application.ScreenUpdating = False

    currentDate = startDate
    Do While currentDate <= endDate
        Call CreateCalendarSheet(currentDate, holidays)
        ' 翌月へ
        currentDate = DateSerial(Year(currentDate), Month(currentDate) + 1, 1)
    Loop

    Application.ScreenUpdating = True

    MsgBox "カレンダーの作成が完了しました！", vbInformation, "カレンダー作成"

End Sub

'==============================================================================
' サブルーチン: CreateCalendarSheet
' 引数:
'   targetDate  - 対象月の1日 (Date型)
'   holidays    - 祝日辞書 (Scripting.Dictionary)
' 概要: 1か月分のカレンダーシートを新規作成・整形する
'==============================================================================
Private Sub CreateCalendarSheet(targetDate As Date, holidays As Object)

    Dim ws          As Worksheet    ' 新規シート
    Dim sheetName   As String       ' シート名（例: 2024年10月）
    Dim yr          As Integer      ' 年
    Dim mo          As Integer      ' 月
    Dim firstDay    As Date         ' その月の1日
    Dim lastDay     As Date         ' その月の末日
    Dim startCol    As Integer      ' 1日が属する曜日列（日=1, 月=2,...土=7）
    Dim dayNum      As Integer      ' 日付カウンタ
    Dim totalDays   As Integer      ' 月の総日数
    Dim curRow      As Integer      ' 書き込み行
    Dim curCol      As Integer      ' 書き込み列
    Dim i           As Integer
    Dim dateKey     As String       ' 祝日チェック用キー

    yr = Year(targetDate)
    mo = Month(targetDate)
    sheetName = yr & "年" & mo & "月"

    '--------------------------------------------------------------------------
    ' 同名シートが既に存在する場合の処理
    '--------------------------------------------------------------------------
    Dim existingWs As Worksheet
    For Each existingWs In ThisWorkbook.Worksheets
        If existingWs.Name = sheetName Then
            Dim answer As VbMsgBoxResult
            answer = MsgBox( _
                "シート「" & sheetName & "」は既に存在します。" & vbCrLf & _
                "上書きしますか？", _
                vbYesNo + vbQuestion, "シート重複")
            If answer = vbYes Then
                Application.DisplayAlerts = False
                existingWs.Delete
                Application.DisplayAlerts = True
            Else
                Exit Sub  ' このシートはスキップ
            End If
            Exit For
        End If
    Next existingWs

    '--------------------------------------------------------------------------
    ' 新規シートを末尾に追加
    '--------------------------------------------------------------------------
    Set ws = ThisWorkbook.Worksheets.Add( _
        After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
    ws.Name = sheetName

    '--------------------------------------------------------------------------
    ' 月の基本情報を計算
    '--------------------------------------------------------------------------
    firstDay = DateSerial(yr, mo, 1)
    lastDay  = DateSerial(yr, mo + 1, 0)  ' 翌月0日 = 今月末日
    totalDays = Day(lastDay)

    ' 1日の曜日を取得（VBA Weekday: 1=日, 2=月, ..., 7=土）
    ' カレンダー列は A(日)=1, B(月)=2, ..., G(土)=7 とする
    startCol = Weekday(firstDay, vbSunday)  ' 1(日)〜7(土)

    '==========================================================================
    ' レイアウト設計
    '   行1   : 月タイトル（例: 2024年10月）
    '   行2   : 曜日ヘッダー（日 月 火 水 木 金 土）
    '   行3〜 : カレンダー本体（最大6週 = 6行）
    '==========================================================================

    '--------------------------------------------------------------------------
    ' ① タイトル行（行1）
    '--------------------------------------------------------------------------
    With ws.Range("A1:G1")
        .Merge
        .Value = sheetName
        .Font.Size = 18
        .Font.Bold = True
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .Interior.Color = RGB(31, 73, 125)   ' 濃紺
        .Font.Color = RGB(255, 255, 255)      ' 白文字
    End With
    ws.Rows(1).RowHeight = 36

    '--------------------------------------------------------------------------
    ' ② 曜日ヘッダー行（行2）
    '--------------------------------------------------------------------------
    Dim dayNames As Variant
    dayNames = Array("日", "月", "火", "水", "木", "金", "土")

    For i = 1 To 7
        With ws.Cells(2, i)
            .Value = dayNames(i - 1)
            .Font.Bold = True
            .Font.Size = 12
            .HorizontalAlignment = xlCenter
            .VerticalAlignment = xlCenter
            Select Case i
                Case 1  ' 日曜
                    .Interior.Color = RGB(255, 230, 230)  ' 薄赤
                    .Font.Color = RGB(192, 0, 0)           ' 濃赤
                Case 7  ' 土曜
                    .Interior.Color = RGB(218, 232, 252)  ' 薄青
                    .Font.Color = RGB(31, 73, 125)         ' 濃青
                Case Else
                    .Interior.Color = RGB(242, 242, 242)  ' 薄グレー
                    .Font.Color = RGB(0, 0, 0)
            End Select
        End With
    Next i
    ws.Rows(2).RowHeight = 24

    '--------------------------------------------------------------------------
    ' ③ カレンダー本体の書き込み
    '--------------------------------------------------------------------------
    curRow = 3
    curCol = startCol

    For dayNum = 1 To totalDays

        '--- 日付キー（祝日チェック用）
        dateKey = Format(yr, "0000") & "/" & Format(mo, "00") & "/" & Format(dayNum, "00")

        '--- セル書き込み
        With ws.Cells(curRow, curCol)
            .Value = dayNum
            .HorizontalAlignment = xlCenter
            .VerticalAlignment = xlTop
            .Font.Size = 13
            .Font.Bold = True

            '--- 祝日チェック（日曜・祝日 = 赤、土曜 = 青）
            If holidays.Exists(dateKey) Then
                ' 祝日
                .Font.Color = RGB(192, 0, 0)
                .Interior.Color = RGB(255, 245, 245)

                ' 祝日名をセルの下部に小さく表示（コメントで補完も可）
                Dim adjCell As Range
                Set adjCell = ws.Cells(curRow, curCol)
                adjCell.Value = dayNum & Chr(10) & holidays(dateKey)
                adjCell.WrapText = True
                adjCell.Font.Size = 10

            ElseIf curCol = 1 Then
                ' 日曜
                .Font.Color = RGB(192, 0, 0)
                .Interior.Color = RGB(255, 245, 245)

            ElseIf curCol = 7 Then
                ' 土曜
                .Font.Color = RGB(31, 73, 125)
                .Interior.Color = RGB(240, 247, 255)

            Else
                ' 平日
                .Font.Color = RGB(0, 0, 0)
                .Interior.Color = RGB(255, 255, 255)
            End If

        End With

        '--- 次の列へ。土曜（列7）の次は日曜（列1）の次の行
        curCol = curCol + 1
        If curCol > 7 Then
            curCol = 1
            curRow = curRow + 1
        End If

    Next dayNum

    '--------------------------------------------------------------------------
    ' ④ セルの枠線・行高・列幅の整形
    '--------------------------------------------------------------------------
    ' カレンダー全体の最終行
    Dim lastCalRow As Integer
    lastCalRow = curRow
    If curCol = 1 Then lastCalRow = curRow - 1  ' 最後が行頭の場合

    ' 枠線をカレンダーエリア全体に設定
    Dim calRange As Range
    Set calRange = ws.Range(ws.Cells(2, 1), ws.Cells(lastCalRow, 7))
    With calRange.Borders
        .LineStyle = xlContinuous
        .Weight = xlThin
        .Color = RGB(180, 180, 180)
    End With

    ' 外枠を太く
    With calRange.BorderAround(xlContinuous, xlMedium, RGB(100, 100, 100))
    End With

    ' 行の高さ設定（カレンダー本体）
    Dim r As Integer
    For r = 3 To lastCalRow
        ws.Rows(r).RowHeight = 60
    Next r

    ' 列幅を均等に
    For i = 1 To 7
        ws.Columns(i).ColumnWidth = 13
    Next i

    '--------------------------------------------------------------------------
    ' ⑤ 余白・印刷設定
    '--------------------------------------------------------------------------
    With ws.PageSetup
        .Orientation = xlLandscape
        .FitToPagesWide = 1
        .FitToPagesTall = 1
        .PaperSize = xlPaperA4
        .TopMargin    = Application.InchesToPoints(0.5)
        .BottomMargin = Application.InchesToPoints(0.5)
        .LeftMargin   = Application.InchesToPoints(0.5)
        .RightMargin  = Application.InchesToPoints(0.5)
        .CenterHorizontally = True
    End With

    ' シート全体のフォントをメイリオに統一（読みやすさのため）
    ws.Cells.Font.Name = "メイリオ"

    ' タイトルだけフォント再設定（上書き回避）
    ws.Range("A1").Font.Size = 18
    ws.Range("A1").Font.Bold = True
    ws.Range("A1").Font.Color = RGB(255, 255, 255)
    ws.Range("A1").Font.Name = "メイリオ"

End Sub

'==============================================================================
' 関数: IsValidYearMonth
' 引数: input_ - ユーザー入力文字列
' 戻値: True = 正しい YYYY/MM 形式、False = 不正
'==============================================================================
Private Function IsValidYearMonth(input_ As String) As Boolean
    IsValidYearMonth = False

    ' 長さチェック（"YYYY/MM" = 7文字）
    If Len(input_) <> 7 Then Exit Function

    ' スラッシュ位置チェック
    If Mid(input_, 5, 1) <> "/" Then Exit Function

    ' 数字チェック
    Dim yrStr As String, moStr As String
    yrStr = Left(input_, 4)
    moStr = Right(input_, 2)

    If Not IsNumeric(yrStr) Or Not IsNumeric(moStr) Then Exit Function

    ' 月の範囲チェック（01〜12）
    Dim moNum As Integer
    moNum = CInt(moStr)
    If moNum < 1 Or moNum > 12 Then Exit Function

    ' 年の範囲チェック（1900〜2100）
    Dim yrNum As Integer
    yrNum = CInt(yrStr)
    If yrNum < 1900 Or yrNum > 2100 Then Exit Function

    IsValidYearMonth = True
End Function
